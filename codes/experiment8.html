from pyspark.sql import SparkSession
spark = SparkSession.builder.appName("PySpark Data Visualization").master("local[*]").getOrCreate()
!wget https://jacobcelestine.com/knowledge_repo/colab_and_pyspark/cars.csv
df = spark.read.csv("cars.csv", header=True, sep=";")
df.show(5, truncate=False)
df.printSchema()
from pyspark.sql.functions import col
print("TOTAL RECORD COUNT:", df.count())
print("EUROPE FILTERED RECORD COUNT:", df.filter(col("Origin")=="Europe").count())
df.filter(col("Origin")=="Europe").show(truncate=False)
df.orderBy("Cylinders").show(truncate=False)
